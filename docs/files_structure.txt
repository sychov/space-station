Files and foldres structure of the source project.
(чтобы понимать логику распределения)
===========================================


 - environment                       (- игровое окружение -)
    \
        - objects                        (- неподвижные, но активные объекты на карте -)
            \
              storages                       (- объекты, обладающие свойствами "хранилища" -)
			      \
				   base_storage.py               (базовый класс хранилища)
				   metal_locker.py               (металлический шкаф)
				   wooden_locker.py              (деревянный шкаф)

			  doors                          (- объекты-"двери"  -)
			      \
					simple_door.py                (простая горизонтальная дверь (зеленая). Открывается вручную.)
					smart_door.py                 ("умная" горизонтальная дверь (синяя). Открывается через терминал)
					door_terminal.py              (дверной терминал)
					smart_armored_door.py         ("бронированная умная" дверь (лиловая). Открывается через терминал)
					gate_door.py                  (дверь парного шлюза. Открывается через специальный терминал шлюза)
					gate_terminal.py              (терминал щлюза)
			  
			  base.py                        (базовый объект)
			  _types_table.py                (таблица соотношений тексовых обозначений классов в файлах настроек игры с самими классами)
			  

        - items                   (- предметы -)
            \
              base.py                 (базовый предмет. Класс будет сильно дорабатываться, пока времянка)
              ...                     (... в перспективе много других предметов, основанных на базовом)

        map.py                    (игровая карта)
        map_cells.py              (классы клеток на игровой карте)
        objects_manager.py        (связка между картой и активными объектами на ней)
        storage_container.py      (упорядоченный контейнер для вещей)
        _tile_collisions.py       (справочник коллизий клеток для объектов на карте)


 - chars                          (- персонажи игроков, а также иные подвижные объекты без привязки к карте -)
     \
	  chars_manager.py                (менеджер всех персонажей)
      base.py                         (базовый персонаж в игре)
      player.py                       (непосредственный игрок)


 - interface                      (- разнообразные интерфейсы, рисуемые поверх игрового поля -)
    \
    - storages                    (- интерфейсы хранилищ -)
        \
        base.py                       (базовый интерфейс "окошка"-хранилища)
        metal_locker.py               (интерфейс металлического ящика)
        wooden_box.py                 (интерфейс деревянного ящика)
		pseudo_inventory.py           (имитатор инвентаря игрока. Сугубая времянка)
        ...                           (... много других подобных интерфейсов, основанных на базовом)

    hud.py                        (менеджер всего интерфейса)
    frame.py                      (базовый класс "окошка" в игре)
    frame_manager.py              (менеджер "окошек")
    log.py                        ("окошко" для вывода лога игроввых событий)
    actions.py                    (кнопки для действий игрока - и да, где-то рядом потом родится базовый класс кнопки)

 - sounds                     (- обеспечение работы звуков и музыки -)
     \
    sound_library.py              (библиотека игровых звуков)
	music_box.py                  (библиотека игровой музыки)

- misc                        (- всякое разное -)
    \
    events.py                     (кастомные игровые ивенты)
    _enums.py                     (enumы - бессмысленные константы с внятными именами для использования в качестве ключей и параметров вместо строк)
    _pathes.py                    (пути к файлам в игре)

main.py                       (точка входа)
